class FreeSourceValidator {
  validateSource(source: Source): number {
    let score = 0.5;

    const govDomains = ['irs.gov', 'gov.in', 'ustaxcourt.gov', 'cbic.gov.in'];
    if (govDomains.some(d => source.url?.includes(d))) score = 0.95;
    
    // Court cases (high trust)
    if (source.type === 'case' && source.court) score = 0.90;
    
    // Official publications
    if (source.type === 'publication' && source.official) score = 0.85;
    
    // Check cross-references (cited by others)
    const citationCount = this.countCitations(source.id);
    if (citationCount > 10) score += 0.05;
    if (citationCount > 50) score += 0.05;
    
    // Recency (newer = more relevant for tax)
    const age = Date.now() - new Date(source.date).getTime();
    const yearsSincePublished = age / (365 * 24 * 60 * 60 * 1000);
    if (yearsSincePublished < 2) score += 0.05;
    
    return Math.min(score, 1.0);
  }
}

class FreeCitationGraph {
  async extractCitations(documentText: string) {
    // Regex patterns for common citation formats
    const patterns = [
      /\d+\s+U\.S\.\s+\d+/g,              // US Supreme Court
      /\d+\s+T\.C\.\s+\d+/g,              // Tax Court
      /IRC\s+§\s*\d+/g,                   // Internal Revenue Code
      /Section\s+\d+[A-Za-z]*/g,          // General sections
    ];
    
    const citations = [];
    for (const pattern of patterns) {
      const matches = documentText.match(pattern) || [];
      citations.push(...matches);
    }

  }

        await db.query(`
          INSERT INTO document_citations (from_doc_id, to_citation, weight)
          VALUES ($1, $2, 1)
          ON CONFLICT DO NOTHING
        `, [doc.id, citation]);
      }
    }
  }
}
